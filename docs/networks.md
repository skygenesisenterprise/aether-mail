# Réseaux externes requis pour le déploiement avec Traefik et Dokploy

# Créer les réseaux externes avant le déploiement :

# docker network create aethermail-aethermail

# docker network create dokploy_local-network

# Le réseau aethermail-aethermail permet la communication entre les services de l'application

# Le réseau dokploy_local-network permet la communication avec Traefik et les autres services Dokploy
